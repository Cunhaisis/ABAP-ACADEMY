REPORT zclientes0602.

TABLES kna1.

*&---------------------------------------------------------------------*
*
*&---------------------------------------------------------------------*
TYPES: BEGIN OF ty_kna1,
        kunnr  TYPE kunnr,
        name1  TYPE name1_gp,
        ort01  TYPE ort01_gp,
        regio  TYPE regio,
        stras  TYPE stras,
        END OF ty_kna1.

*&---------------------------------------------------------------------*
*Workarea
*&---------------------------------------------------------------------*
DATA: wa_kna1 TYPE ty_kna1.
*&---------------------------------------------------------------------*
*Tabela Interna
*&---------------------------------------------------------------------*
DATA: ti_kna1 TYPE STANDARD TABLE OF ty_kna1.

*&---------------------------------------------------------------------*
*Selection-Screeen
*&---------------------------------------------------------------------*
SELECT-OPTIONS s_kunnr FOR kna1-kunnr.

*&---------------------------------------------------------------------*
*Start-of-Selection
*&---------------------------------------------------------------------*
START-OF-SELECTION.

  SELECT kunnr
         name1
         ort01
         regio
         stras
    FROM kna1
    INTO TABLE ti_kna1
    WHERE kunnr IN s_kunnr.

  LOOP AT ti_kna1 INTO wa_kna1.
    WRITE: / wa_kna1-kunnr,
             wa_kna1-name1,
             wa_kna1-ort01,
             wa_kna1-regio,
             wa_kna1-stras.
  ENDLOOP.
